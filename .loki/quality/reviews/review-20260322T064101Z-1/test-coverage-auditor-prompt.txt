+      break

+      store.setLineup(sampleLineup)

+      break
+    }

+    case 'COMPLETE_ACT': {
+      const s = useShowStore.getState()
+      if (s.currentActId) {

+        // Skip beat (don't wait for celebration timeout)
+        useShowStore.getState().skipBeat()

+      break

+    case 'ENTER_INTERMISSION':
+      store.enterIntermission()
+      break
+
+    case 'EXIT_INTERMISSION':
+      store.exitIntermission()
+      break
+
+    case 'ENTER_DIRECTOR':
+      store.enterDirector()
+      break
+
+    case 'EXIT_DIRECTOR':
+      store.exitDirector()
+      break
+
+    case 'CALL_EARLY':

+      break
+
+    case 'SKIP_TO_NEXT':
+      store.skipToNextAct()
+      break
+
+    case 'STRIKE':

+      break

+      break
